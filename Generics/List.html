def self.[](type)
  Class.new(self).tap do |klass|
    klass.instance_variable_set('@type', type)
    klass.instance_variable_set('@type_checker', TypeChecker[type])
    klass.define_singleton_method(:name) { klass.superclass.name }
  end
end
